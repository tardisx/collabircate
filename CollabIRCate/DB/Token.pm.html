sub new_link_token {
    my $class    = shift;
    my $irc_user = shift;

    croak "called as object method" if ( ref $class );

    my $md5 = Digest::MD5->new();
    $md5->add($$);
    $md5->add( time() );
    $md5->add($irc_user);

    my $self = __PACKAGE__->new(
        token   => $md5->hexdigest,
        expires => time() + $expires,
        type => 'link',
        data    => $irc_user,
    );

    return $self;
}

Create a token to link an IRC user to a real user.

sub new_upload_token {
    my $class       = shift;
    my $channel_id  = shift;    # a channel id
    my $irc_user_id = shift;    # an IRC user id

    croak "called as object method" if ( ref $class );

    my $md5 = Digest::MD5->new();
    $md5->add($$);
    $md5->add( time() );
    $md5->add($channel_id);
    $md5->add($irc_user_id);

    my $self = __PACKAGE__->new(
        token   => $md5->hexdigest,
        expires => time() + $expires,
        type    => 'upload',
        data    => "$channel_id|$irc_user_id",
    );

    return $self;
}

Creates a token for a file upload. Potential file uploads require an IRC user and a channel destination for the upload.
