NAME

CollabIRCate::Bot::Plugin::Google - allow the bot to search google for you.

DESCRIPTION

This plugin allows the bot to answer requests like 'google foo'. The response is the top hit for that Google search request.

sub answer {
  my $class = shift;
  my $question = shift;
  return unless ( $question =~ /google\s+(.*)/i );

  my $q = $1;
  $q =~ s/[^\w\s]//g;

  # set service to use
  REST::Google->service(
    'http://ajax.googleapis.com/ajax/services/search/web');

  # provide a valid http referer
  REST::Google->http_referer('http://collabircate.eatmorecode.com');

  my $res = REST::Google->new( q => $q );

  return [ 'google response failure', undef ]
    if $res->responseStatus != 200;

  my $data = $res->responseData;

  my $result =
    $data->{results}->[0]->{url} . ' | '
      . $data->{results}->[0]->{content};

  $result =~ s{}{}g;

  return { answer => $result };

}
