sub answer {
    my $user    = shift;
    my $channel = shift;
    my $msg     = shift;

#    die "who $who where $channel msg $msg";

    if ( $msg
        =~ /^(what\s*[i']s\s:{0,1}){0,1}\s*([\d\+\-\s\*\/\.\,]+)([\s\=\?]+){0,1}$/
        )
    {
        $msg =~ s/[^\d\+\-\*\/\^\s\.]//g;
        my $answer = eval "$msg";

        my $response = CollabIRCate::Bot::Response->new;
        if (!$@) {
            $response->add_response({channel => $channel,
                                     user => $user,
                                     text => "the answer is $answer"});
        }
        else {
            $response->add_response({channel => $channel,
                                     user => $user,
                                     text => "nice try, $msg is not valid"});
        }
        return $response;
    }
    else {
        # not a math question
        return;
    }


}
